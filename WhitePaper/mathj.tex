\begin{lstlisting}[]
library(rgl)
source('Function.R')

# Matrix size.
size = c(3, 4)

# Example tree drawing.
values =  c(-1, -2, -1, -2, 1, -3, 2, -4, 1, -3)
merges = MergeMatrix(size, values)
heights = c(1, 1.3, 4, 7, 10)
rowcols = c(0, 1, 1, 1, 0)

# Plot forestogramme.
Forestogramme(size, merges, heights, rowcols)
\end{lstlisting}
